   > **修复说明**：原计划在 `ConfigureServices` 中调用 `BuildServiceProvider()` 提取 Options，这是 .NET 绝对禁止的 DI 反模式，会导致生命周期错乱。已修正为直接从 `context.Configuration` 读取字符串。
```csharp
using DeviceSimulator.Infrastructure;
using IoTHunter.Shared.Infrastructure;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Hosting;
using Serilog;
var host = Host.CreateDefaultBuilder(args)
    .UseSerilog((context, services, configuration) => {
        LoggingDefaults.ConfigureBaseLogger(configuration);
    })
    .ConfigureServices((context, services) => {
        services.Configure<SimulatorOptions>(context.Configuration.GetSection("Simulator"));
        
        // 🚨 终审修复：绝对禁止在 ConfigureServices 中调用 BuildServiceProvider()
        var httpBase = context.Configuration["Simulator:HttpBaseUrl"] ?? "http://localhost:5080";
        
        services.AddHttpClient("Gateway", client => {
            client.BaseAddress = new Uri(httpBase);
            client.Timeout = TimeSpan.FromSeconds(10);
        });
        
        services.AddIoTHunterOpenTelemetry(context.Configuration, "DeviceSimulator");
        services.AddHostedService<SimulatorWorker>();
    })
    .Build();
await host.RunAsync();
```

## 4.4 子步骤四：MQTT 发送器
**目标**：将 `TelemetryEnvelope` 构造成 MQTT 消息，严格对齐网关契约与 QoS 策略。
### 4.4.1 动作项
1. **🚨 终审修复：新建 `DeviceSimulator/Infrastructure/MqttTelemetrySender.cs`**
   > **修复说明**：原计划未给出具体 Payload 序列化代码。如果直接序列化 `TelemetryEnvelope`（含 PascalCase 和 DateTimeOffset），网关反序列化必败，100% 进入 DLQ。已修复为严格构造网关契约格式，并强制对齐 ADR-014（QoS 分级）和 ADR-016（随机 ClientId）。
```csharp
using IoTHunter.Shared.Domain;
using MQTTnet;
using MQTTnet.Client;
using System.Text;
using System.Text.Json;
namespace DeviceSimulator.Infrastructure;
internal sealed class MqttTelemetrySender : IAsyncDisposable
{
    private readonly ILogger<MqttTelemetrySender> _logger;
    private IMqttClient? _client;
    public MqttTelemetrySender(ILogger<MqttTelemetrySender> logger)
    {
        _logger = logger;
    }
    public async Task EnsureConnectedAsync(string broker, int port, string clientIdPrefix, CancellationToken ct)
    {
        if (_client?.IsConnected == true) return;
        var factory = new MqttFactory();
        _client = factory.CreateMqttClient();
        
        // 🚨 对齐 ADR-016：普通压测必须使用随机 ClientId
        var clientId = $"{clientIdPrefix}-{Guid.NewGuid():N}";
        
        var options = new MqttClientOptionsBuilder()
            .WithTcpServer(broker, port)
            .WithClientId(clientId)
            .WithCleanSession(false) 
            .Build();
        await _client.ConnectAsync(options, ct);
        _logger.LogInformation("MQTT Simulator connected as {ClientId}", clientId);
    }
    public async Task<SendResult> SendAsync(TelemetryEnvelope envelope, CancellationToken ct)
    {
        if (_client == null || !_client.IsConnected) return SendResult.FatalFailure;
        var isCritical = envelope.ReliabilityLevel == ReliabilityLevel.Critical;
        var topic = isCritical 
            ? $"device/{envelope.DeviceId}/event/critical" 
            : $"device/{envelope.DeviceId}/telemetry";
        
        // 🚨 终极修复：MQTT Payload 必须严格等于第二步网关反序列化的 TelemetryRequest 结构！
        var payload = JsonSerializer.Serialize(new {
            deviceId = envelope.DeviceId,
            metricType = envelope.MetricType,
            timestamp = envelope.RecordedAt.ToUnixTimeMilliseconds(),
            sequence = envelope.Sequence
        });
        var message = new MqttApplicationMessageBuilder()
            .WithTopic(topic)
            .WithPayload(payload)
            // 🚨 对齐 ADR-014：关键事件必须 QoS 1，普通遥测 QoS 0
            .WithQualityOfServiceLevel(isCritical 
                ? MQTTnet.Protocol.MqttQualityOfServiceLevel.AtLeastOnce 
                : MQTTnet.Protocol.MqttQualityOfServiceLevel.AtMostOnce)
            .Build();
        try
        {
            var result = await _client.PublishAsync(message, ct);
            return SendResult.Success;
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "MQTT publish failed {EventId}", envelope.EventId);
            return SendResult.RetryableFailure;
        }
    }
    public async ValueTask DisposeAsync()
    {
        if (_client != null)
        {
            await _client.DisconnectAsync();
            _client.Dispose();
        }
    }
}
```

```powershell
Write-Host "=== Step 4 E2E Verification ===" -ForegroundColor Cyan
# 1. 启动 Simulator (建议直接修改 appsettings.json 中的 Protocol 为 http 后运行)
Write-Host "Starting Simulator for 30 seconds..." -ForegroundColor Yellow
dotnet run --project DeviceSimulator/DeviceSimulator.csproj
# 2. 等待 Redis 投影追平 (ADR-005 规定投影存在延迟)
Write-Host "Waiting 3 seconds for Redis projection..." -ForegroundColor DarkYellow
Start-Sleep -Seconds 3
# 3. 查询 BackendProcessor API
Write-Host "=== API Query Verification ===" -ForegroundColor Yellow
try {
    $latest = Invoke-RestMethod -Uri "http://localhost:5081/api/v2/devices/dev-001/latest" -ErrorAction Stop
    Write-Host "Latest device dev-001 found: $($latest | ConvertTo-Json -Compress)" -ForegroundColor Green
} catch {
    if ($_.Exception.Response.StatusCode.value__ -eq 404) {
        Write-Host "Latest API returned 404 (This is CORRECT behavior per ADR, projection not ready yet)" -ForegroundColor DarkYellow
    } else {
        Write-Host "API Error: $($_.Exception.Message)" -ForegroundColor Red
    }
}
$history = Invoke-RestMethod -Uri "http://localhost:5081/api/v2/devices/dev-001/history?page=1&pageSize=5"
Write-Host "History total count for dev-001: $($history.totalCount)" -ForegroundColor Green
Write-Host "=== Step 4 Verification Complete ===" -ForegroundColor Cyan
```
